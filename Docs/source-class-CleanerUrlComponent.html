41: class CleanerUrlComponent extends Component {
42:     /**
43:      * Called before the controller's beforeFilter method. Execute cleanUrl
44:      *
45:      * Look at {@link http://api.cakephp.org/2.4/class-Component.html#_initialize CakePHP Api}
46:      * @param Controller $controller
47:      */
48:     public function initialize(Controller $controller) {
49:         //If there's a query string
50:         if(!empty($_SERVER["QUERY_STRING"]))
51:             $this->cleanUrl($controller);
52:     }
53: 
54:     /**
55:      * Clean the current url, removing the query string, then executes redirect
56:      * @param Controller $controller
57:      */
58:     protected function cleanUrl(Controller $controller) {
59:         //Take the query string and traslate it in the $queryString array
60:         parse_str($_SERVER["QUERY_STRING"], $queryString);
61:         //Delete all empty values, then recreate the query string (now without empty values)
62:         $queryString = http_build_query(array_filter($queryString));
63:         //Change "&" with "/" and "=" with ":" from the query string
64:         $queryString = str_replace(array('&', '='), array('/', ':'), $queryString);
65: 
66:         /**
67:          * Perform the redirect
68:          *
69:          * $controller->request->here contains the current url, to which is appended the cleaned query string
70:          */
71:         $controller->redirect($controller->request->here.'/'.$queryString);
72:     }
73: }
