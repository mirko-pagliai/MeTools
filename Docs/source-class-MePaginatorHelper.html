 31:  * Provides functionalities to the generation of pagers.
 32:  * 
 33:  * Rewrites {@link http://api.cakephp.org/2.5/class-PaginatorHelper.html PaginatorHelper}.
 34:  *

 37:  * public $helpers = array('Paginator' => array('className' => 'MeTools.MePaginator'));
 38:  * </code>

 40: class MePaginatorHelper extends PaginatorHelper {
 41:     /**
 42:      * Helpers
 43:      * @var array 
 44:      */
 45:     public $helpers = array('Html' => array('className' => 'MeTools.MeHtml'));
 46: 
 47:     /**
 48:      * Internal function for jump links.
 49:      * 
 50:      * For example:
 51:      * <code>
 52:      * list($title, $options, $disabledTitle, $disabledOptions) = $this->_jump_link($title, $options, $disabledTitle, $disabledOptions);
 53:      * </code>
 54:      * @param string $title Link title
 55:      * @param array $options Options for pagination link
 56:      * @param string $disabledTitle Title when the link is disabled
 57:      * @param array $disabledOptions Options for the disabled pagination link
 58:      * @return array Passed arguments
 59:      * @uses MeHtmlHelper::_addOptionDefault()
 60:      * @uses MeHtmlHelper::_addOptionValue()
 61:      */
 62:     private function _jump_link($title = NULL, $options = array(), $disabledTitle = NULL, $disabledOptions = array()) {
 63:         //If the "disabled title" is empty, it will be set to the title
 64:         $disabledTitle = empty($disabledTitle) ? $title : $disabledTitle;
 65: 
 66:         $options = $this->Html->_addOptionDefault('escape', FALSE, $options);
 67:         $options = $this->Html->_addOptionDefault('tag', 'li', $options);
 68: 
 69:         $disabledOptions = $this->Html->_addOptionValue('class', 'disabled', $disabledOptions);
 70:         $disabledOptions = $this->Html->_addOptionDefault('disabledTag', 'a', $disabledOptions);
 71:         $disabledOptions = $this->Html->_addOptionDefault('tag', 'li', $disabledOptions);
 72: 
 73:         return array($title, $options, $disabledTitle, $disabledOptions);
 74:     }
 75: 
 76:     /**
 77:      * Returns a counter string for the paged result set.
 78:      * 
 79:      * By default, returns something like "1 - 20 of 50", which means that you're viewing records 1 to 20 of 50 total.
 80:      * @param array $options Options for the counter string
 81:      * @return string Counter string
 82:      */
 83:     public function counter($options = array()) {
 84:         //"format" option default "{:start} - {:end} of {:count}"
 85:         $options = $this->Html->_addOptionDefault('format', __d('me_tools', '%s - %s of %s', '{:start}', '{:end}', '{:count}'), $options);
 86: 
 87:         return parent::counter($options);
 88:     }
 89: 
 90:     /**
 91:      * Returns the counter string as disabled link.
 92:      * @param array $options Options for the counter string
 93:      * @return string Counter string as disabled link
 94:      * @uses counter()
 95:      * @uses MeHtmlHelper::li()
 96:      * @uses MeHtmlHelper::link()
 97:      */
 98:     public function counterLink($options = array()) {
 99:         return $this->Html->li($this->Html->link(self::counter($options), '#'), array('class' => 'disabled'));
100:     }
101: 
102:     /**
103:      * Generates a "next" link for a set of paged records.
104:      * @param string $title Link title
105:      * @param array $options Options for pagination link
106:      * @param string $disabledTitle Title when the link is disabled
107:      * @param array $disabledOptions Options for the disabled pagination link
108:      * @return string A "next" link or $disabledTitle text if the link is disabled
109:      * @uses _jump_link()
110:      * @uses MeHtmlHelper::_addOptionValue()
111:      */
112:     public function next($title = NULL, $options = array(), $disabledTitle = NULL, $disabledOptions = array()) {
113:         if(empty($disabledOptions) && !empty($options)) {
114:             $disabledOptions = $options;
115:             $disabledOptions = $this->Html->_addOptionValue('class', 'disabled', $disabledOptions);
116:         }
117:         
118:         //Uses `self::__jump->link()` to set arguments
119:         list($title, $options, $disabledTitle, $disabledOptions) = self::_jump_link($title, $options, $disabledTitle, $disabledOptions);
120: 
121:         return parent::next($title, $options, $disabledTitle, $disabledOptions);
122:     }
123: 
124:     /**
125:      * Returns a set of numbers for the paged result set uses a modulus to decide how many numbers .
126:      * to show on each side of the current page (default: 8).
127:      * @param array $options Options for the numbers.
128:      * @return string Numbers string
129:      * @uses MeHtmlHelper::_addOptionDefault()
130:      * @uses MeHtmlHelper::_addOptionValue()
131:      */
132:     public function numbers($options = array()) {
133:         $options = $this->Html->_addOptionValue('currentClass', 'active', $options);
134:         $options = $this->Html->_addOptionDefault('currentTag',  'a', $options);
135:         $options = $this->Html->_addOptionDefault('separator', FALSE, $options);
136:         $options = $this->Html->_addOptionDefault('tag', 'li', $options);
137: 
138:         return parent::numbers($options);
139:     }

141:     /**
142:      * Generates a "prev" link for a set of paged records.
143:      * @param string $title Link title
144:      * @param array $options Options for pagination link
145:      * @param string $disabledTitle Title when the link is disabled
146:      * @param array $disabledOptions Options for the disabled pagination link
147:      * @return string A "prev" link or $disabledTitle text if the link is disabled
148:      * @uses _jump_link()
149:      * @uses MeHtmlHelper::_addOptionValue()
150:      */
151:     public function prev($title = NULL, $options = array(), $disabledTitle = NULL, $disabledOptions = array()) {
152:         if(empty($disabledOptions) && !empty($options)) {
153:             $disabledOptions = $options;
154:             $disabledOptions = $this->Html->_addOptionValue('class', 'disabled', $disabledOptions);
155:         }
156:         
157:         //Uses self::__jump->link() to set arguments
158:         list($title, $options, $disabledTitle, $disabledOptions) = self::_jump_link($title, $options, $disabledTitle, $disabledOptions);
159: 
160:         return parent::prev($title, $options, $disabledTitle, $disabledOptions);
161:     }
