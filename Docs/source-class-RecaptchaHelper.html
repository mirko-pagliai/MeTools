 32: /**
 33:  * Provides several methods for reCAPTCHA.
 34:  *
 35:  * Before using this helper, you have to configure keys in `app/Config/recaptcha.php`.
 36:  * You can use as example the file `app/Plugin/MeTools/Config/recaptcha.default.php`.
 37:  */
 38: class RecaptchaHelper extends AppHelper {
 39:     /**
 40:      * Mail keys
 41:      * @var array
 42:      */
 43:     private $mail_keys = FALSE;
 44: 
 45:     /**
 46:      * Helpers
 47:      * @var array
 48:      */
 49:     public $helpers = array('Html' => array('className' => 'MeTools.MeHtml'));
 50: 
 51:     /**
 52:      * Construct
 53:      * @param View $View The View this helper is being attached to
 54:      * @param array $settings Configuration settings for the helper
 55:      * @uses mail_keys to set mail keys
 56:      */
 57:     public function __construct(View $View, $settings = array()) {
 58:         Configure::load('recaptcha');
 59: 
 60:         $keys = $this->mail_keys = array(
 61:            'pub'    => Configure::read('Recaptcha.Mail.Public_key'),
 62:            'priv'   => Configure::read('Recaptcha.Mail.Private_key')
 63:         );
 64: 
 65:         if(empty($keys['pub']) || empty($keys['priv']))
 66:             throw new InternalErrorException(__d('me_tools', 'Mail keys are not configured'));
 67: 
 68:         //Checks if the private mail key is valid (hexadecimal digits)
 69:         if(!ctype_xdigit($keys['priv']))
 70:             throw new InternalErrorException(__d('me_tools', 'The private mail key is not valid'));
 71: 
 72:         parent::__construct($View, $settings);
 73:     }
 74: 
 75:     /**
 76:      * Alias for `mailLink()` method
 77:      * @see mailLink()
 78:      */
 79:     public function mail() {
 80:         return call_user_func_array(array('RecaptchaHelper', 'mailLink'), func_get_args());
 81:     }

 83:     /**
 84:      * Creates an HTML link for an hidden email. The link will be open in a popup.
 85:      * @param string $title Link title
 86:      * @param string $mail Email to hide

 88:      * @return string Html

 92:      */
 93:     public function mailLink($title, $mail, $options = array()) {
 94:         $link = $this->mailUrl($mail);
 95:

 97:         $options['onclick'] = sprintf("window.open('%s', '', 'toolbar=0,scrollbars=0,location=0,statusbar=0,menubar=0,resizable=0,width=500,height=300'); return false;", $link);
 98: 
 99:         return $this->Html->link($title, $link, $options);
100:     }

102:     /**
103:      * Gets the url for an hidden email. 
104:      * 
105:      * This method will only return a url. If you want to create a link, you should use the `mailLink()` method
106:      * @param string $mail Email to hide
107:      * @return string Url
108:      * @see mailLink()
109:      * @see http://repository.novatlantis.it/metools-sandbox/recaptcha/maillinks Examples
110:      * @uses mail_keys to read mail keys
111:      */
112:     public function mailUrl($mail) {
113:         return !empty($this->mail_keys) ? recaptcha_mailhide_url($this->mail_keys['pub'], $this->mail_keys['priv'], $mail) : NULL;
114:     }
