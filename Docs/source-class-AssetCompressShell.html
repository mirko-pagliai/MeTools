 33: /**
 34:  * This shell allows you to combine and compress css and js files.
 35:  * 
 36:  * To use this shell, you have to install on your system `Clean-css` and `UglifyJS`. As root user:
 37:  * <code>
 38:  * npm install clean-css -g
 39:  * npm install uglify-js -g
 40:  * </code>
 41:  */
 42: class AssetCompressShell extends MeToolsAppShell {
 43:     /**
 44:      * Parses arguments or a config file, checks values and returns input and output files
 45:      * @return array Input and output files
 46:      */
 47:     private function _parse() {
 48:         //Checks if there are at least 2 arguments or a config file
 49:         if(count($this->args) < 2 && empty($this->params['config']))
 50:             $this->error('you have to indicate at least two arguments, an input file and an output file, or a configuration file by using the option');
 51:         
 52:         //If it needs to use a configuration file
 53:         if(!empty($this->params['config'])) {
 54:             //Checks if the config file is readable
 55:             if(!is_readable($config = $this->params['config']))
 56:                 $this->error(sprintf('%s doesn\'t exists or is not readable', $config));
 57:             
 58:             //Sets each line of the configuration file as an argument
 59:             $this->args = explode(PHP_EOL, file_get_contents($config));
 60:             
 61:             //Checks if there are at least 2 files
 62:             if(count($this->args))
 63:                 $this->error('you have to indicate at least two arguments, an input file and an output file');
 64:         }
 65:         
 66:         //Gets the output file and the input files. The last argument is the output file, the other arguments are the input files
 67:         $output = array_pop($this->args);
 68:         $input = is_array($this->args) ? $this->args : array($this->args);
 69:                 
 70:         //Checks that all input files exist and are readable 
 71:         foreach($input as $file) {
 72:             if(!is_readable($file))
 73:                 $this->error(sprintf('%s doesn\'t exists or is not readable', $file));
 74:         }
 75:         
 76:         //If the output file already exists and has not been used the "force" option, it asks if it should be overwritten
 77:         if(file_exists($output) && empty($this->params['force']) && $this->in(sprintf('The file %s already exists. Do you want to overwrite it?', $output), array('y', 'n'), 'y') === 'n') {
 78:             $this->out('Ok, i\'m exiting...');
 79:             exit;
 80:         }
 81:         
 82:         //Returns input and output files
 83:         return array($input, $output);
 84:     }
 85:     
 86:     /**
 87:      * Combines and compresses css files
 88:      * @uses _parse()
 89:      * @uses System::which()
 90:      */
 91:     public function css() {
 92:         //Checks for Clean-css
 93:         if(!($cleancss = System::which('cleancss')))
 94:             $this->error(sprintf('I can\'t find %s', 'Clean-css'));
 95:         
 96:         //Gets the output file and the input files
 97:         list($input, $output) = $this->_parse();
 98:         
 99:         //Executes the command
100:         exec(sprintf('%s -o %s %s', $cleancss, $output, implode(' ', $input)));
101:     }
102:     
103:     /**
104:      * Combine and compresses js files
105:      * @uses _parse()
106:      * @uses System::which()
107:      */
108:     public function js() {
109:         if(!($uglifyjs = System::which('uglifyjs')))
110:             $this->error(sprintf('I can\'t find %s', 'UglifyJS'));
111:         
112:         //Gets the output file and the input files
113:         list($input, $output) = $this->_parse();
114:         
115:         //Executes the command
116:         exec(sprintf('%s --compress --mangle -o %s %s', $uglifyjs, $output, implode(' ', $input)));
117:     }
118:     
119:     /**
120:      * Gets the option parser instance and configures it.
121:      * @return ConsoleOptionParser
122:      */
123:     public function getOptionParser() {
124:         $parser = parent::getOptionParser();
125:         $parser->addSubcommand('css', array(
126:             'help' => 'Combines and compresses css files'
127:         ))->addSubcommand('js', array(
128:             'help' => 'Combines and compresses js files'
129:         ))->addOption('config', array(
130:             'help'  => 'Configuration file',
131:             'short' => 'c'
132:         ))->addOption('force', array(
133:             'boolean'   => TRUE,
134:             'help'      => 'Force overwriting existing files without prompting',
135:             'short'     => 'f'
136:         ));
137:         
138:         return $parser;
139:     }
140: }
