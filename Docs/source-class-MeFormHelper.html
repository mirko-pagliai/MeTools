 41:      * Creates a simple button. Rewrites <i>$this->Form->button()</i>
 42:      * @param string $caption The button label or an image
 43:      * @param array $options Options
 44:      * @return string Html
 45:      */
 46:     public function button($caption, $options=array()) {
 47:         //"type" option default "button"
 48:         $options['type'] = empty($options['type']) ? 'button' : $options['type'];
 49: 
 50:         //"class" option default "btn"
 51:         $options['class'] = empty($options['class']) ? 'btn' : $this->Html->cleanAttribute($options['class'].' btn');
 52: 
 53:         //Adds an icon to the label, if the "icon" option exists
 54:         $caption = !empty($options['icon']) ? $this->Html->icon($options['icon']).$caption : $caption;
 55:         unset($options['icon']);
 56: 
 57:         //Adds the "tooltip" rel
 58:         $options['rel'] = empty($options['rel']) ? 'tooltip' : $this->Html->cleanAttribute($options['rel'].' tooltip');
 59: 
 60:         return parent::button($caption, $options);

 64:      * Closes a form. Rewrites <i>$this->Form->end()</i> and use the <i>button()</i> method
 65:      *
 66:      * If $options is set, a submit button will be created. Options can be either a string or an array
 67:      * @param string $caption The submit button label or an image
 68:      * @param array $options Options
 69:      * @return string Html

 71:     public function end($caption=null, $options=null) {
 72:         //Unsets the "label" option
