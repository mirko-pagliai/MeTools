140:     /**

142:      * @param string $path Path
143:      * @return boolean TRUE if they are readable and writable, FALSE otherwise
144:      */
145:     public static function dirIsWritable($directory) {
146:         if(!is_readable($directory))
147:             return FALSE;
148:         
149:         $folder = new Folder();
150: 
151:         foreach($folder->tree($directory, FALSE, 'dir') as $dir) {
152:             if(!is_readable($dir) || !is_writable($dir))
153:                 return FALSE;
154:         }

156:         return TRUE;

208:     /**
209:      * Gets the version number for each plugin.
210:      * @param string|array $except Plugins to exclude
211:      * @return mixed array with the version number for each plugin
212:      */
213:     public static function getPluginsVersion($except = NULL) {
214:         //Gets plugins
215:         $plugins = CakePlugin::loaded();
216:         
217:         //Removes the exceptions from the list
218:         if(is_string($except) || is_array($except)) {
219:             $except = is_array($except) ? $except : array($except);
220:             $plugins = array_diff($plugins, $except);
221:         }
222:         
223:         //For each plugin, gets the name and the version number
224:         $pluginsTmp = array();
225:         foreach($plugins as $plugin) {
226:             if(is_readable($file = CakePlugin::path($plugin).'version'))
227:                 $pluginsTmp[] = array('name' => $plugin, 'version' => trim(file_get_contents($file)));
228:         }
229:         
230:         return $pluginsTmp;
231:     }
