 28: /**
 29:  * An utility for checking the status of the system and perform maintenance tasks.
 30:  * 
 31:  * You can use this utility by adding in your controller:
 32:  * <code>
 33:  * App::uses('System', 'MeTools.Utility');
 34:  * </code>
 35:  */
 36: class System {
 37:     /**
 38:      * Checks if a directory and its subdirectories are readable and writable
 39:      * @param string $path Path
 40:      * @return boolean TRUE if they are readable and writable, FALSE otherwise
 41:      */
 42:     private static function _dirIsWritable($directory) {
 43:         if(!is_readable($directory))
 44:             return FALSE;
 45:         
 46:         $folder = new Folder();
 47: 
 48:         foreach($folder->tree($directory, FALSE, 'dir') as $dir) {
 49:             if(!is_readable($dir) || !is_writable($dir))
 50:                 return FALSE;
 51:         }
 52: 
 53:         return TRUE;
 54:     }
 55: 
 56: 
 57:     /**
 58:      * Checks if an Apache module is active
 59:      * @param string $module Name of the module to be checked
 60:      * @return boolean TRUE if the module is enabled, FALSE otherwise
 61:      * @uses getApacheModules() to get Apache's modules
 62:      */
 63:     public static function checkApacheModule($module) {
 64:         return in_array($module, self::getApacheModules());
 65:     }
 66: 
 67:     /**
 68:      * Checks if the cache and all its subdirectories are readable and writable
 69:      * @return boolean TRUE if if the cache is readable and writable, FALSE otherwise
 70:      * @uses _dirIsWritable to check if is readable and writable
 71:      */
 72:     public static function checkCache() {
 73:         return self::_dirIsWritable(CACHE);
 74:     }

 86:      * @param string $required_version Required version of PHP

 88:      * @uses getPhpVersion() to get the current version of PHP
 89:      */
 90:     public static function checkPhpVersion($required_version) {
 91:         return version_compare(self::getPhpVersion(), $required_version, '>=');
 92:     }

129:     /**
130:      * Clears thumbnails
131:      * @return boolean TRUE if thumbnails were successfully cleared, FALSE otherwise
132:      */
133:     public static function clearThumbs() {
134:         $dir = new Folder($tmp = TMP.'thumbs');
135:         $files = $dir->findRecursive('.+\..+');
136:         $success = TRUE;
137:         
138:         foreach($files as $file) {
139:             $file = new File($file);
140:             if(!$file->delete() && $success)
141:                 $success = FALSE;
142:         }
143:         
144:         return $success;
145:     }

214:     /**
215:      * Executes the `whereis` command on Unix systems.
216:      * 
217:      * It locates the binary files for a command.
218:      * @param string $command Command
219:      * @return mixed Array of binary files, otherwise FALSE
220:      */
221:     public static function whereis($command) {
222:         $whereis = explode(' ', exec(sprintf('whereis -b %s', $command)));
223:             
224:         unset($whereis[0]);

226:         return empty($whereis) ? FALSE : $whereis;
