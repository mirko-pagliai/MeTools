 31: /**
 32:  * Creates and displays thumbnails for image and video files.
 33:  * 
 34:  * The `thumb()` action takes the maximum width and/or the maximum height as query string parameters (`w` and `h` parameters).
 35:  * It can also create square thumbs. In this case, it's sufficient to indicate the maximum side in the query string (`s` parameter).
 36:  * With square thumbs, the initial image will be cut off if it is rectangular.
 37:  * 
 38:  * You can set the maximum height only for image files.
 39:  * Instead, the video thumbnails can be created using the maximum width or maximum side.
 40:  * 
 41:  * `ThumbsController` doesn't just show thumbnails, but creates real thumbnails in the temporary directory (`app/tmp/thumbs`), 
 42:  * which can be used later when the same thumbs will be required (as if it were a cache).
 43:  * 
 44:  * To display a thumbnail or get the url for a thumbnail, you have to use `thumb()` or `thumbUrl()` methods provided by the `MeHtml` helper.
 45:  * The `thumb()` method, using this controller, creates the thumbnail and returns the HTML code to show it.
 46:  * The `thumbUrl()` method creates the thumbs and returns its url.
 47:  * @see MeHtmlHelper::thumb(), MeHtmlHelper::thumbUrl()
 48:  */
 49: class ThumbsController extends MeToolsAppController {
 50:     /**
 51:      * Creates a thumbnail of an image.
 52:      * @param object $file File object
 53:      * @return string Thumbnail path, if a thumbnail has been created
 54:      * @throws InternalErrorException
 55:      */
 56:     private function _imageThumb($file) {
 57:         //Checks for Imagick
 58:         if(!extension_loaded('imagick'))
 59:             throw new InternalErrorException(__d('me_tools', 'The %s library is missing', 'Imagick'));
 60:         
 61:         //Gets the maximum sizes
 62:         $maxWidth = (int) $this->request->query('w');
 63:         $maxHeight = (int) $this->request->query('h');
 64:         $maxSide = (int) $this->request->query('s');
 65:         
 66:         //If no size is specified, then it's not necessary to create a thumbnail
 67:         if(!$maxWidth && !$maxHeight && !$maxSide)
 68:             return $file->path;
 69:         
 70:         //Creates the Imagick object
 71:         $image = new Imagick($file->path);
 72:         
 73:         //If the max side is defined (then has been requested a square thumb)
 74:         if($maxSide) {
 75:             //If the maximum side is larger than the width and height, then the maximum side is equal to the smallest size
 76:             if($maxSide > $image->getImageWidth() || $maxSide > $image->getImageHeight())
 77:                 $maxSide = $image->getImageWidth() > $image->getImageHeight() ? $image->getImageHeight() : $image->getImageWidth();
 78:                         
 79:             //Creates the thumbnail
 80:             $image->cropThumbnailImage($finalWidth = $finalHeight = $maxSide, $maxSide);
 81:         }
 82:         //Else, if the maximum width and the maximum height are defined
 83:         elseif($maxWidth && $maxHeight) {
 84:             //Tries to get the final sizes from the width
 85:             $finalWidth = floor($image->getImageWidth() * $maxHeight / $image->getImageHeight());
 86: 
 87:             //If the final width is greater than the maximum width, it gets the final sizes from the final height
 88:             if($finalWidth > $maxWidth)
 89:                 $finalHeight = floor($image->getImageHeight() * ($finalWidth = $maxWidth) / $image->getImageWidth());
 90:             //Else, the final height is the maximum height
 91:             else
 92:                 $finalHeight = $maxHeight;
 93:         
 94:             //Creates the thumbnail
 95:             $image->thumbnailImage($finalWidth, $finalHeight);
 96:         }
 97:         //Else, if only the maximum width is defined
 98:         elseif($maxWidth) {
 99:             //If the maximum width is greater than the actual width, then it's not necessary to create a thumbnail
100:             if(($finalWidth = $maxWidth) >= $image->getImageWidth())
101:                 return $file->path;
102:             
103:             //Gets the final height and creates the thumbnail
104:             $finalHeight = floor($finalWidth * $image->getImageHeight() / $image->getImageWidth());
105:             $image->thumbnailImage($finalWidth, 0);
106:             
107:         }
108:         //Else, if only the maximum height is defined
109:         else {
110:             //If the maximum height is greater than the actual height, then it's not necessary to create a thumbnail
111:             if(($finalHeight = $maxHeight) >= $image->getImageHeight())
112:                 return $file->path;
113:             
114:             //Gets the final width and creates the thumbnail
115:             $finalWidth = floor($finalHeight * $image->getImageWidth() / $image->getImageHeight());
116:             $image->thumbnailImage(0, $finalHeight);
117:         }
118:         
119:         //Gets the thumbnail path
120:         $thumb = TMP.'thumbs'.DS.'photos'.DS.md5($file->path).'_'.$finalWidth.'x'.$finalHeight.'.jpg';
121:         
122:         //Checks if the thumbnail already exists
123:         if(is_readable($thumb))
124:             return $thumb;
125:         
126:         //Checks if the target directory is writable
127:         if(!is_writable(dirname($thumb)))
128:             throw new InternalErrorException(__d('me_tools', 'The target directory %s is not writable', dirname($thumb)));
129:         
130:         //Writes the image to the output directory and destroys the Imagick object
131:         $image->writeImage($thumb);
132:         $image->destroy();
133:         
134:         //Checks if the thumbnail has been created
135:         if(!is_readable($thumb))
136:             throw new InternalErrorException(__d('me_tools', 'The thumbnail %s has not been created', $thumb));
137:         
138:         return $thumb;
139:     }
140:     
141:     /**
142:      * Creates a thumbnail of a video.
143:      * @param object $file File object
144:      * @return string Thumbnail path, if a thumbnail has been created
145:      * @throws InternalErrorException
146:      */
147:     private function _videoThumb($file) {
148:         //Gets the maximum sizes
149:         $maxWidth = (int) $this->request->query('w');
150:         $maxSide = (int) $this->request->query('s');
151:         
152:         //If no size is specified, it sets a maximum width
153:         if(!$maxWidth && !$maxSide)
154:             $maxWidth = 270;
155:         
156:         //Gets the thumbnail path
157:         if($maxSide)
158:             $thumb = TMP.'thumbs'.DS.'videos'.DS.md5($file->path).'_s'.($maxWidth = $maxSide).'.jpg';
159:         else
160:             $thumb = TMP.'thumbs'.DS.'videos'.DS.md5($file->path).'_w'.$maxWidth.'.jpg';
161:                 
162:         //Checks if the thumbnail already exists
163:         if(is_readable($thumb))
164:             return $thumb;
165:         
166:         //Checks for ffmpegthumbnailer
167:         if(empty(System::which('ffmpegthumbnailer')))
168:             throw new InternalErrorException(__d('me_tools', '%s is not avalaible', 'ffmpegthumbnailer'));
169:         
170:         //Checks if the target directory is writable
171:         if(!is_writable(dirname($thumb)))
172:             throw new InternalErrorException(__d('me_tools', 'The target directory %s is not writable', dirname($thumb)));
173:         
174:         //Creates the thumbnail
175:         shell_exec($cmd = sprintf('ffmpegthumbnailer -s %s -q 10 -f -i \'%s\' -o \'%s\'', $maxWidth, $file->path, $thumb));
176:         
177:         //Checks if the thumbnail has been created
178:         if(!is_readable($thumb))
179:             throw new InternalErrorException(__d('me_tools', 'The thumbnail %s has not been created', $thumb));
180:         
181:         //If the max side is defined (then has been requested a square thumb)
182:         if($maxSide) {
183:             //Creates the Imagick object
184:             $image = new Imagick($thumb);
185:             
186:             //If the height of the thumbnail is larger than the width
187:             if($image->getImageHeight() < $image->getImageWidth()) {
188:                 //Adds a border
189:                 $border = ($image->getImageWidth() - $image->getImageHeight()) / 2;
190:                 $image->borderImage('#000000', 0, $border);
191:                 
192:                 //Writes the image to output directory and destroys the Imagick object
193:                 $image->writeImage($thumb);
194:                 $image->destroy();
195:             }
196:         }
197:         
198:         return $thumb;
199:     }

201:     /**
202:      * Creates and shows thumbnails for images and video.
203:      * 
204:      * Please, refer to the class description for more information.
205:      * It's convenient to use `thumb()` or `thumbUrl()` method provided by the `MeHtml` helper.
206:      * @param string $file File path, encoded by `base64_encode()`
207:      * @throws InternalErrorException
208:      * @throws NotFoundException
209:      * @see MeHtmlHelper::thumb(), MeHtmlHelper::thumbUrl()
210:      * @uses _imageThumb() to create a thumbnail of an image
211:      * @uses _videoThumb() to create a thumbnail of a video
212:      */
213:     public function thumb($file = FALSE) {
214:         //Checks the file path
215:         if(empty($file))
216:             throw new InternalErrorException(__d('me_tools', 'The file has not been specified'));
217: 
218:         //Decodes the path. If the path is relative, then it's relative to the webroot
219:         $file = urldecode(base64_decode($file));
220:         $file = !Folder::isAbsolute($file) ? WWW_ROOT.$file : $file;
221: 
222:         //Checks if the file is readable
223:         if(!is_readable($file))
224:             throw new NotFoundException(__d('me_tools', 'The file %s doesn\'t exist or is not readable', $file));
225:         
226:         //Now `$file` is the File object
227:         $file = new File($file);
228:         $mime = $file->mime();
229: 
230:         //If the file is an image
231:         if(preg_match('/image\/\S+/', $mime))
232:             $thumb = $this->_imageThumb($file);
233:         //Else, if the file is a video
234:         elseif(preg_match('/video\/\S+/', $mime) || $mime == 'application/ogg')
235:             $thumb = $this->_videoThumb($file);
236:         //Else, if the mime type is not known
237:         else
238:             throw new InternalErrorException(__d('me_tools', 'The mime type %s is not supported', $mime));
239:         
240:         //Renders the thumbnail
241:         header("Content-type: image/jpeg");
242:         readfile($thumb);
243:         
244:         $this->autoRender = FALSE;
245:         exit;
246:     }
