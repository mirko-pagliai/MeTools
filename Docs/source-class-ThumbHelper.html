 4: /**
 5:  * ThumbHelper allows you to generate thumbnails of images, using {@link https://code.google.com/p/phpthumbmaker Thumber}.
 6:  *
 7:  * Before using ThumbHelper, the file <i>Plugin/MeTools/webroot/thumber.php</i> must be copied into <i>webroot/</i> (the webroot of your app).
 8:  *
 9:  * This helper just returns the url of the thumb, then <i>thumber.php</i> will really create the thumb.
10:  *
11:  * The thumbnails will be created in <i>webroot/files/thumbs/</i> (the webroot of your app), which must be writable.
12:  *
13:  * Example, in your view:
14:  * <code>
15:  * echo $this->Html->image($this->Thumb->getUrl('/path/to/image.jpg', 200));

41: class ThumbHelper extends MeToolsAppHelper {
42:     /**
43:      * Get the url of a photo thumbnail
44:      * @param string $path Image path
45:      * @param string|int $width Thumb width
46:      * @param string|int $height Thumb height
47:      * @return string Thumb url
48:      */
49:     public function getUrl($path, $width=null, $height=null) {
50:         //Initial url
51:         $url = '/thumber.php?img='.urlencode(base64_encode($path));
52: 
53:         //If requested to resize according to the width and to the height
54:         if(!empty($width) && !empty($height)) {
55:             //Get image sizes
56:             $originalSizes = getimagesize($path);
57: 
58:             //If width is larger or equal than height, append width to the url
59:             if($originalSizes['0'] >= $originalSizes['1'])
60:                 $url .= '&w='.$width;
61:             //Else, if height is larger than height, append height to the url
62:             else
63:                 $url .= '&h='.$height;
64:         }
65:         //If requested to resize according to the width, append width to the url
66:         elseif(!empty($width))
67:             $url .= '&w='.$width;
68:         //If requested to resize according to the height, append height to the url
69:         elseif(!empty($height))
70:             $url .= '&h='.$height;
71: 
72:         return $url;
73:     }
74: }
