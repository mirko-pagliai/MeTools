310:     public function fixComposerJson() {
311:         if(!is_writeable($file = ROOT.DS.'composer.json'))
312:             return $this->err(__d('me_tools', 'File or directory `{0}` not writeable', rtr($file)));
313:         
314:         //Gets and decodes the file
315:         $contents = json_decode(file_get_contents($file), TRUE);
316:         
317:         //Checks if the file has been fixed
318:         if(!empty($contents['config']['component-dir']) && $contents['config']['component-dir'] === 'vendor/components')
319:             return $this->verbose(__d('me_tools', 'The file `{0}` doesn\'t need to be fixed', rtr($file)));
320:                 
321:         //Fixeds and encodes the content
322:         $contents = (new File($file))->prepare(json_encode(am($contents, ['config' => ['component-dir' => 'vendor/components']]), JSON_UNESCAPED_SLASHES | JSON_PRETTY_PRINT));
323:         
324:         if((new File($file))->write($contents))
325:             $this->verbose(__d('me_tools', 'The file `{0}` has been fixed', rtr($file)));
326:         else
327:             $this->err(__d('me_tools', 'The file `{0}` has not been fixed', rtr($file)));

356:      * Install the suggested packages
357:      * @param bool $force
358:      * @uses MeTools\Utility\Unix::which()
359:      * @uses $packages
360:      */
361:     public function installPackages($force = FALSE) {
362:         //Checks for Composer
363:         if(!($bin = Unix::which('composer')))
364:             return $this->err(__d('me_tools', '{0} is not available', 'composer'));
365:         
366:         //Empty array. This will contain the packages to install
367:         $packagesToInstall = [];
368:         
369:         //Asks whick packages to install, if it was not asked to install all of them or if you are not using the "force" parameter
370:         if(!$force && !$this->param('force')) {
371:             foreach($this->packages as $package) {
372:                 $ask = $this->in(__d('me_tools', 'Do you want to install `{0}`?', $package), ['Y', 'n'], 'Y');
373:                 if(in_array($ask, ['Y', 'y']))
374:                     $packagesToInstall[] = $package;
375:             }
376:         
377:             if(empty($packagesToInstall))
378:                 return $this->verbose(__d('me_tools', 'No package has been selected for installation'));
379:         }
380:         else
381:             $packagesToInstall = $this->packages;
382:         
383:         //Gets the list of installed packages
384:         exec(sprintf('%s show --installed', $bin), $installed);
385:         $installed = array_map(function($line) { return array_values(preg_split('/[\s]+/', $line))[0]; }, $installed);
386:         
387:         $packagesToInstall = array_diff($packagesToInstall, $installed);
388:         
389:         if(empty($packagesToInstall))
390:             return $this->verbose(__d('me_tools', 'All packages are already installed'));
391:             
392:         //Executes the command
393:         exec(sprintf('%s require %s', $bin, implode(' ', $packagesToInstall)));
394:     }
