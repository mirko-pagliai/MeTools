 30: /**
 31:  * A component to check for reCAPTCHA.
 32:  *
 33:  * Before using this comoonent, you have to configure keys in `app/Config/recaptcha.php`.
 34:  * You can use as example the file `app/Plugin/MeTools/Config/recaptcha.default.php`.
 35:  */
 36: class RecaptchaComponent extends Component {
 37:     /**
 38:      * Controller
 39:      * @var object
 40:      */
 41:     protected $controller;
 42:     
 43:     /**
 44:      * Error
 45:      * @var string 
 46:      */
 47:     protected $error = NULL;
 48:     
 49:     /**
 50:      * reCAPTCHA form keys
 51:      * @var array Keys
 52:      */
 53:     private $keys = array();
 54:     
 55:     /**
 56:      * Constructor.
 57:      * @param ComponentCollection $collection A ComponentCollection this component can use to lazy load its components
 58:      * @param array $settings Array of configuration settings.
 59:      * @uses error
 60:      */
 61:     public function __construct(ComponentCollection $collection, $settings = array()) {
 62:         parent::__construct($collection, $settings);
 63:         
 64:         //Sets the error message
 65:         $this->error = __d('me_tools', 'You have not filled out the %s control', 'reCAPTCHA');
 66:     }
 67:     
 68:     /**
 69:      * Is called after the controller's beforeFilter method but before the controller executes the current action handler.
 70:      * @param Controller $controller
 71:      * @uses controller
 72:      * @uses keys
 73:      */
 74:     public function startup(Controller $controller) {
 75:         //Loads the configuration file
 76:         Configure::load('recaptcha');
 77:         
 78:         //Gets form keys
 79:         $this->keys = Configure::read('Recaptcha.Form');
 80:         
 81:         //Checks for form keys
 82:         if(empty($this->keys['public']) || empty($this->keys['private']))
 83:             throw new InternalErrorException(__d('me_tools', 'Form keys are not configured'));
 84:         
 85:         $this->controller = $controller;
 86:     }
 87:     
 88:     /**
 89:      * Checks for reCAPTCHA.
 90:      * @return bool TRUE on success, otherwise FALSE
 91:      * @see https://developers.google.com/recaptcha/docs/verify
 92:      * @uses controller
 93:      * @uses error
 94:      * @uses keys
 95:      * @uses HttpSocket::post()
 96:      */
 97:     public function check() {
 98:         //Checks for response
 99:         if(empty($this->controller->request->data['g-recaptcha-response'])) {
100:             $this->error = __d('me_tools', 'You have not filled out the %s control', 'reCAPTCHA');  
101:             return FALSE;
102:         }
103:         else
104:             $response = $this->controller->request->data['g-recaptcha-response'];
105:         
106:         //Post request
107:         $http = new HttpSocket();
108:         $results = $http->post('https://www.google.com/recaptcha/api/siteverify', am(array(
109:             'remoteip'  => $this->controller->request->clientIp(TRUE),
110:             'secret'    => $this->keys['private']
111:         ), compact('response')));
112:                 
113:         if(empty($results) || !$results = json_decode($results, TRUE) || empty($results['success'])) {
114:             $this->error = __d('me_tools', 'It was not possible to verify the %s control', 'reCAPTCHA');
115:             return FALSE;
116:         }
117:         
118:         return TRUE;
119:     }
120:     
121:     /**
122:      * Gets the last error.
123:      * @return mixed Error or NULL
124:      * @uses error
125:      */
126:     public function getError() {
127:         return $this->error;
128:     }
129: }
