 26: use Cake\View\Helper;

 28: /**
 29:  * Provides functionalities for creating dropdown menus, according to Bootstrap.
 30:  * 
 31:  * The `menu()` method creates a full dropdown menu, with a link to open the menu and the menu itself.
 32:  * 
 33:  * Otherwise you can use the `button()` or the `link()`method, which generate a link or a button to open the menu, 
 34:  * followed by the `dropdown()` method, which generates only the menu.
 35:  */
 36: class DropdownHelper extends Helper {
 37:     /**
 38:      * Helpers
 39:      * @var array
 40:      */
 41:     public $helpers = ['Html' => ['className' => 'MeTools.Html']];
 42:     
 43:     /**
 44:      * Parses and handles title and options used to create a link or a button to open a dropdown.
 45:      * 
 46:      * You should not use this method directly, but `button()` or `link()`.
 47:      * @param string $title Link/button title
 48:      * @param array $options HTML attributes and options
 49:      * @return array Array with title and options
 50:      * @see button(), link()
 51:      * @uses MeTools\View\Helper\HtmlHelper::icon()
 52:      */
 53:     protected function __parseLink($title, array $options = []) {
 54:         $title = sprintf('%s %s', $title, $this->Html->icon('caret-down'));
 55:         
 56:         $options = addValue('class', 'dropdown-toggle', $options);
 57:         $options = addValue('data-toggle', 'dropdown', $options);
 58: 
 59:         return [$title, $options];
 60:     }
 61:     
 62:     /**
 63:      * Creates a button to open a dropdown menu, according to Bootstrap.
 64:      * 
 65:      * Note that this method creates only a link. To create a full dropdown menu, you should use the `menu()` method.
 66:      * @param string $title Button title
 67:      * @param array $options Array of options and HTML attributes
 68:      * @return string Html code
 69:      * @see menu()
 70:      * @see http://getbootstrap.com/components/#dropdowns Bootstrap documentation
 71:      * @uses MeTools\View\Helper\HtmlHelper::button()
 72:      * @uses __parseLink()
 73:      */
 74:     public function button($title, array $options = []) {
 75:         //Backward compatibility, if they were passed 3 arguments
 76:         $options = func_num_args() === 3 ? func_get_arg(2) : $options;
 77: 
 78:         list($title, $options) = self::__parseLink($title, $options);
 79: 
 80:         return $this->Html->button($title, '#', $options);
 81:     }
 82:     
 83:     /**
 84:      * Creates a dropdown menu.
 85:      * 
 86:      * Note that this method creates only a dropdown submenu, without the a link or a button to open the menu.
 87:      * To create a full dropdown menu, you should use the `menu()` method.
 88:      * @param array $links Array of links for the dropdown (you should use the `HtmlHelper::link()` method for each link)
 89:      * @param array $options Options for the dropdown (`<ul>` element)
 90:      * @param array $itemOptions Options for each item (`<li>` element)
 91:      * @return string Html code
 92:      * @see menu()
 93:      * @see http://getbootstrap.com/components/#dropdowns Bootstrap documentation
 94:      * @uses MeTools\View\Helper\HtmlHelper::ul()
 95:      */
 96:     public function dropdown(array $links = [], array $options = [], array $itemOptions = []) {
 97:         $options = addValue('class', 'dropdown-menu', $options);
 98:         $options = addValue('role', 'menu', $options);
 99:         $itemOptions = addValue('role', 'presentation', $itemOptions);
100:                 
101:         return $this->Html->ul($links, $options, $itemOptions);
102:     }
103:     
104:     /**
105:      * Creates a link to open a dropdown menu, according to Bootstrap.
106:      * 
107:      * Note that this method creates only a link. To create a full dropdown menu, you should use the `menu()` method.
108:      * @param string $title Link title
109:      * @param array $options Array of options and HTML attributes
110:      * @return string Html code
111:      * @see menu()
112:      * @see http://getbootstrap.com/components/#dropdowns Bootstrap documentation
113:      * @uses MeTools\View\Helper\HtmlHelper::link()
114:      * @uses __parseLink()
115:      */
116:     public function link($title, array $options = []) {
117:         //Backward compatibility, if they were passed 3 arguments
118:         $options = func_num_args() === 3 ? func_get_arg(2) : $options;
119:         
120:         list($title, $options) = self::__parseLink($title, $options);
121: 
122:         return $this->Html->link($title, '#', $options);
123:     }
124:     
125:     /**
126:      * Creates a full menu, according to Bootstrap. For example:
127:      * <code>
128:      * <div class="dropdown">
129:      *  <?php
130:      *      echo $this->Dropdown->menu('Open the dropdown', ['icon' => 'fa-bell'], [
131:      *          $this->Html->link('Github', 'http://github.com', ['icon' => 'fa-github']),
132:      *          $this->Html->link('Stack Overflow', 'http://stackoverflow.com', ['icon' => 'fa-stack-overflow'])
133:      *      ]);
134:      *  ?>
135:      * </div>
136:      * </code>
137:      * @param string $title Link title
138:      * @param array $titleOptions Array of options and HTML attributes
139:      * @param array $links Array of links for the dropdown (you should use the `HtmlHelper::link()` method for each link)
140:      * @param array $dropdownOptions Options for the dropdown (`<ul>` element)
141:      * @param array $itemOptions Options for each item (`<li>` element)
142:      * @return string Html code
143:      * @uses dropdown()
144:      * @uses link()
145:      */
146:     public function menu($title, array $titleOptions = [], array $links = [], array $dropdownOptions = [], array $itemOptions = []) {
147:         return implode(PHP_EOL, [
148:             $this->link($title, $titleOptions),
149:             $this->dropdown($links, $dropdownOptions, $itemOptions)
150:         ]);
151:     }
152: }
