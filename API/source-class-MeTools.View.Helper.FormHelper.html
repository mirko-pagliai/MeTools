248:     /**
249:      * Generates an input element complete with label and wrapper div.
250:      * @param string $fieldName Field name, should be "Modelname.fieldname"
251:      * @param array $options HTML attributes and options
252:      * @return string Html code
253:      * @uses MeTools\View\Helper\HtmlHelper::span()
254:      */
255:     public function input($fieldName, array $options = []) {
256:         //Gets the input type
257:         $type = empty($options['type']) ? self::_inputType($fieldName, $options) : $options['type'];
258:         
259:         $options = addValue('class', 'form-control', $options);
260:         
261:         //If the field name contains the word "password", then the field type is "password"
262:         if(preg_match('/password/', $fieldName))
263:             $options = addDefault('type', 'password', $options);
264:         
265:         //Changes the "autocomplete" value from "FALSE" to "off"
266:         if(isset($options['autocomplete']) && !$options['autocomplete'])
267:             $options['autocomplete'] = 'off';
268:         
269:         //If it's a select
270:         if($type === 'select') {
271:             //By default, the `empty` option will be automatically added (with `FALSE` value)
272:             //This option will be used by the `select()` method to see if the option has been added by the user or not
273:             if(!isset($options['empty']))
274:                 $options = addDefault('remove_empty', TRUE, $options);
275:         }
276:         //Else, if it's a textarea
277:         elseif($type === 'textarea') {
278:             $options = addDefault('cols', NULL, $options);
279:             $options = addDefault('rows', NULL, $options);
280:         }
281:         
282:         //Sets the default templates
283:         //These values can be overwritten by other methods
284:         $this->templates([
285:             'formGroup' => '{{label}}{{input}}',
286:             'inputContainer' => '<div class="input form-group {{type}}{{required}}">{{content}}{{tip}}</div>',
287:             'inputContainerError' => '<div class="input form-group {{type}}{{required}} has-error">{{content}}{{tip}}{{error}}</div>',
288:             'nestingLabel' => '{{hidden}}<label{{attrs}}>{{input}}{{text}}</label>'
289:         ]);
290:         
291:         //If is an inline form
292:         if($this->inline) {
293:             //By default, disables tips and error messages
294:             $this->templates([
295:                 'inputContainer' => '<div class="input form-group {{type}}{{required}}">{{content}}</div>',
296:                 'inputContainerError' => '<div class="input form-group {{type}}{{required}} has-error">{{content}}</div>'
297:             ]);
298:             
299:             //If it is not a checkbox
300:             if($type !== "checkbox") {
301:                 if(empty($options['label']))
302:                     $options['label'] = [];
303:                 elseif(is_string($options['label']))
304:                     $options['label'] = ['text' => $options['label']];
305: 
306:                 $options['label'] = addValue('class', 'sr-only', $options['label']);
307:             }
308:         }
309:         
310:         //Sets tips
311:         if(!empty($options['tip']))
312:             $options['tip'] = implode(PHP_EOL, array_map(function($v) {
313:                 return $this->Html->span(trim($v), ['class' => 'help-block']);
314:             }, is_array($options['tip']) ? $options['tip'] : [$options['tip']]));
315:         
316:         return parent::input($fieldName, $options);
317:     }
