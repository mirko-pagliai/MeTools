 63:      * Creates a symbolic link
 64:      * @param string $origin Origin file or directory
 65:      * @param string $target Target link
 66:      * @return bool
 67:      */
 68:     public function createLink($origin, $target) {
 69:         //Checks if the origin file/directory is readable
 70:         if(!is_readable($origin)) {
 71:             $this->verbose(__d('me_tools', 'File or directory `{0}` not readable', rtr($origin)));
 72:             return FALSE;
 73:         }
 74:         
 75:         //Checks if the target directory is writeable
 76:         if(!is_writable(dirname($target))) {
 77:             $this->err(__d('me_tools', 'File or directory `{0}` not writeable', rtr(dirname($target))));
 78:             return FALSE;
 79:         }
 80:         
 81:         //Checks if the link already exists
 82:         if(file_exists($target)) {
 83:             $this->verbose(__d('me_tools', 'Symbolic link `{0}` already exists', rtr($target)));
 84:             return FALSE;
 85:         }       
 86: 
 87:         //Creates the symbolic link
 88:         if(!@symlink($origin, $target)) {
 89:             $this->err(__d('me_tools', 'Failed to create a symbolic link to `{0}`', rtr($target)));
 90:             return FALSE;
 91:         }
 92:         
 93:         $this->verbose(__d('me_tools', 'Created symbolic link to `{0}`', rtr($target)));
 94:         return TRUE;
 95:     }
