105:      * @return boolean TRUE if the cache is writable and were successfully cleared, FALSE otherwise
106:      * @uses Cake\Cache\Cache::clear()
107:      * @uses Cake\Cache\Cache::clearGroup()
108:      * @uses Cake\Cache\Cache::configured()
109:      * @uses Cake\Cache\Cache::groupConfigs()
110:      */
111:     public static function clearCache() {
112:         $success = TRUE;
113:         
114:         //Cleans all cached values for all cache configurations
115:         foreach(Cache::configured() as $config) {
116:             if(!Cache::clear(FALSE, $config) && $success)
117:                 $success = FALSE;
118:         }
119:         
120:         //Clean all keys from the cache belonging to all group configurations
121:         foreach(Cache::groupConfigs() as $groups) 
122:             foreach($groups as $group) {
123:                 if(!Cache::clearGroup($group) && $success)
124:                     $success = FALSE;
125:             }
126:         
127:         return $success;
128:     }

147:     /**
148:      * Gets all changelog files. 
149:      * 
150:      * It searchs into `ROOT` and all loaded plugins.
151:      * @uses MeTools\Core\Plugin::path()
152:      * @return array Changelog files
153:      * @uses Cake\I18n\I18n::locale()
154:      * @uses MeTools\Core\Plugin::path()
155:      */
156:     public static function getChangelogs() {
157:         $files = af(array_map(function($path) {
158:             //Gets the current locale
159:             $locale = substr(\Cake\I18n\I18n::locale(), 0, 2);
160: 
161:             if(!empty($locale) && is_readable($file = sprintf($path.'CHANGELOG_%s.md', $locale)))
162:                 return str_replace(ROOT.DS, NULL, $file);
163:             elseif(is_readable($file = $path.'CHANGELOG.md'))
164:                 return str_replace(ROOT.DS, NULL, $file);
165:             else
166:                 return FALSE;
167:         }, am([ROOT.DS], Plugin::path())));
168:         
169:         //Re-indexes, starting to 1, and returns
170:         return array_combine(range(1, count($files)), array_values($files));
171:     }
