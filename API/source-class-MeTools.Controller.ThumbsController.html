 33: /**
 34:  * Creates and displays thumbnails for image and video files.
 35:  * 
 36:  * The `thumb()` action takes the maximum width and/or the maximum height as query string parameters (`width` and `height` parameters).
 37:  * It can also create square thumbs. In this case, it's sufficient to indicate the maximum side in the query string (`side` parameter).
 38:  * With square thumbs, the initial image will be cut off if it is rectangular.
 39:  * 
 40:  * You can set the maximum height only for image files.
 41:  * Instead, the video thumbnails can be created using the maximum width or maximum side.
 42:  * 
 43:  * `ThumbsController` doesn't just show thumbnails, but creates real thumbnails in the temporary directory, 
 44:  * which can be used later when the same thumbs will be required (as if it were a cache).
 45:  * 
 46:  * To display a thumbnail or get the url for a thumbnail, you have to use `image()` and `url()` methods provided by `ThumbHelper`.
 47:  * The `ThumbHelper::image()` method, using this controller, creates the thumbnail and returns the HTML code to show it.
 48:  * The `ThumbHelper::url()` method creates the thumbs and returns its url.
 49:  * @see MeTools\View\Helper\ThumbHelper::image(), MeTools\View\Helper\ThumbHelper::url()
 50:  */

 70:     /**
 71:      * Creates a thumbnail from an image file
 72:      * @return void
 73:      * @throws InternalErrorException
 74:      * @uses MeTools\Utility\Php::extension()
 75:      * @uses file
 76:      * @uses sizes
 77:      * @uses thumb
 78:      */
 79:     protected function _imageThumb() {
 80:         //Checks for Imagick extension
 81:         if(!Php::extension('imagick'))
 82:             throw new InternalErrorException(__d('me_tools', 'The {0} library is missing', 'Imagick'));
 83:                 
 84:         //Creates the Imagick object
 85:         $imagick = new \Imagick($this->file->path);
 86:         
 87:         //If the max side is defined (then has been requested a square thumb)
 88:         if($this->sizes['side']) {
 89:             //If the maximum side is larger than the width and height, then the maximum side is equal to the smallest size
 90:             if($this->sizes['side'] > $imagick->getImageWidth() || $this->sizes['side'] > $imagick->getImageHeight())
 91:                 $this->sizes['side'] = $imagick->getImageWidth() > $imagick->getImageHeight() ? $imagick->getImageHeight() : $imagick->getImageWidth();
 92:                         
 93:             //Creates the thumbnail
 94:             $imagick->cropThumbnailImage($finalWidth = $finalHeight = $this->sizes['side'], $this->sizes['side']);
 95:         }
 96:         //Else, if the maximum width and the maximum height are defined
 97:         elseif($this->sizes['width'] && $this->sizes['height']) {
 98:             //Tries to get the final sizes from the width
 99:             $finalWidth = floor($imagick->getImageWidth() * $this->sizes['height'] / $imagick->getImageHeight());
100: 
101:             //If the final width is greater than the maximum width, it gets the final sizes from the final height
102:             if($finalWidth > $this->sizes['width'])
103:                 $finalHeight = floor($imagick->getImageHeight() * ($finalWidth = $this->sizes['width']) / $imagick->getImageWidth());
104:             //Else, the final height is the maximum height
105:             else
106:                 $finalHeight = $this->sizes['height'];
107:         
108:             //Creates the thumbnail
109:             $imagick->thumbnailImage($finalWidth, $finalHeight);
110:         }
111:         //Else, if only the maximum width is defined
112:         elseif($this->sizes['width']) {
113:             //If the maximum width is greater than the actual width, then it's not necessary
114:             //to create a thumbnail and the thumbnail will be the original image
115:             if(($finalWidth = $this->sizes['width']) >= $imagick->getImageWidth()) {
116:                 $this->thumb = $this->file->path;
117:                 return;
118:             }
119:             
120:             //Gets the final height and creates the thumbnail
121:             $finalHeight = floor($finalWidth * $imagick->getImageHeight() / $imagick->getImageWidth());
122:             $imagick->thumbnailImage($finalWidth, 0);
123:             
124:         }
125:         //Else, if only the maximum height is defined
126:         else {
127:             //If the maximum height is greater than the actual height, then it's not necessary
128:             //to create a thumbnail and the thumbnail will be the original image
129:             if(($finalHeight = $this->sizes['height']) >= $imagick->getImageHeight()) {
130:                 $this->thumb = $this->file->path;
131:                 return;
132:             }
133:                 
134:             //Gets the final width and creates the thumbnail
135:             $finalWidth = floor($finalHeight * $imagick->getImageWidth() / $imagick->getImageHeight());
136:             $imagick->thumbnailImage(0, $finalHeight);
137:         }
138:         
139:         //Writes the image to the output directory and destroys the Imagick object
140:         $imagick->writeImage($this->thumb);
141:         $imagick->destroy();
142:         
143:         //Checks if the thumbnail has been created
144:         if(!is_readable($this->thumb))
145:             throw new InternalErrorException(__d('me_tools', 'The thumbnail {0} has not been created', $this->thumb));
146:     }

149:      * Creates a thumbnail from a video file
150:      * @return void

152:      * @uses MeTools\Utility\Unix::which()
153:      * @uses sizes
154:      * @uses thumb

156:     protected function _videoThumb() {
157:         //Checks for ffmpegthumbnailer
158:         if(!Unix::which('ffmpegthumbnailer'))
159:             throw new InternalErrorException(__d('me_tools', '{0} is not avalaible', 'ffmpegthumbnailer'));
160:         
161:         //Creates the thumbnail
162:         shell_exec(sprintf('ffmpegthumbnailer -s %s -q 10 -f -i \'%s\' -o \'%s\'', empty($this->sizes['side']) ? $this->sizes['width'] : $this->sizes['side'], $this->file>path, $this->thumb));
163:         
164:         //Checks if the thumbnail has been created
165:         if(!is_readable($this->thumb))
166:             throw new InternalErrorException(__d('me_tools', 'The thumbnail {0} has not been created', $this->thumb));
167:         
168:         //If the max side is defined (then has been requested a square thumb)
169:         if($this->sizes['side']) {
170:             //Creates the Imagick object
171:             $imagick = new \Imagick($this->thumb);
172:             
173:             //If the height of the thumbnail is larger than the width
174:             if($imagick->getImageHeight() < $imagick->getImageWidth()) {
175:                 //Adds a border
176:                 $imagick->borderImage('#000000', 0, ($imagick->getImageWidth() - $imagick->getImageHeight()) / 2);
177:                 
178:                 //Writes the image to output directory and destroys the Imagick object
179:                 $imagick->writeImage($this->thumb);
180:                 $imagick->destroy();
181:             }
182:         }

185:     /**
186:      * Creates and shows thumbnails for image and video files.
187:      * 
188:      * Please, refer to the class description for more information.
189:      * It's convenient to use `image()` or `url()` methods provided by `ThumbHelper`.
190:      * @param string $file File path, encoded by `base64_encode()`
191:      * @see MeTools\View\Helper\ThumbHelper::image(), MeTools\View\Helper\ThumbHelper::url()
192:      * @throws InternalErrorException
193:      * @uses MeTools\Utility\Thumbs::photo()
194:      * @uses MeTools\Utility\Thumbs::remote()
195:      * @uses MeTools\Utility\Thumbs::video()
196:      * @uses _imageThumb()
197:      * @uses _videoThumb()
198:      * @uses file
199:      * @uses sizes
200:      * @uses thumb
201:      */
202:     public function thumb($file) {
203:         //Removes the fake file extension from the path
204:         $file = pathinfo($file, PATHINFO_FILENAME);
205:         
206:         //Decodes the path
207:         $file = urldecode(base64_decode($file));
208:         
209:         //If the file is remote
210:         if(filter_var($file, FILTER_VALIDATE_URL)) {            
211:             //Downloads the file, if not already done
212:             if(!is_readable($tmp = Thumbs::remote(md5($file).'.'.pathinfo($file, PATHINFO_EXTENSION)))) {
213:                 //Checks if the target directory is writable
214:                 if(!is_writable(dirname($tmp)))
215:                     throw new InternalErrorException(__d('me_tools', 'The directory {0} is not writable', dirname($tmp)));
216:                     
217:                 //Downloads the file
218:                 file_put_contents($tmp, file_get_contents($file));
219:             }
220:             //The file is now the temporary file
221:             $file = $tmp;
222:         }
223:         //Else, if the file is local and its path is relative, then the path will be relative to the webroot
224:         elseif(!Folder::isAbsolute($file))
225:             $file = WWW_ROOT.$file;
226: 
227:         //Checks if the file is readable
228:         if(!is_readable($file))
229:             throw new InternalErrorException(__d('me_tools', 'The file {0} doesn\'t exist or is not readable', $file));
230:         
231:         //Creates the File object
232:         $this->file = new File($file);
233:         
234:         //Sets the maximum sizes        
235:         foreach(['side', 'width', 'height'] as $v)
236:             $this->sizes[$v] = (int) $this->request->query($v);
237:         
238:         //If the file is an image
239:         if(preg_match('/image\/\S+/', $mime = $this->file->mime())) {
240:             //If no maximum size is specified, it's not necessary to create a thumbnail
241:             if(empty(array_filter($this->sizes))) {
242:                 //Renders the original image
243:                 header(sprintf('Content-type: %s', $mime));
244:                 readfile($this->file->path);
245: 
246:                 $this->autoRender = FALSE;
247:                 exit;
248:             }
249:             
250:             //Sets the thumb path
251:             $this->thumb = Thumbs::photo(md5($this->file->path));
252:             
253:             if($this->sizes['side'])
254:                 $this->thumb = sprintf('%s_s_%s', $this->thumb, $this->sizes['side']);
255:             else {
256:                 if($this->sizes['width'])
257:                     $this->thumb = sprintf('%s_w_%s', $this->thumb, $this->sizes['width']);
258:                 if($this->sizes['height'])
259:                     $this->thumb = sprintf('%s_h_%s', $this->thumb, $this->sizes['height']);
260:             }
261:             
262:             //Sets the thumbnail path
263:             $this->thumb = sprintf('%s.%s', $this->thumb, $this->file->ext());
264:         }
265:         //Else, if the file is a video
266:         elseif(preg_match('/video\/\S+/', $mime) || $mime == 'application/ogg') {
267:             if(!$this->sizes['side'] && !$this->sizes['width'])
268:                 $this->sizes['side'] = 270;
269:             
270:             //Sets the thumb path
271:             $this->thumb = Thumbs::video(md5($this->file->path));
272:             
273:             if($this->sizes['side'])
274:                 $this->thumb = sprintf('%s_s_%s', $this->thumb, $this->sizes['side']);
275:             elseif($this->sizes['width'])
276:                 $this->thumb = sprintf('%s_w_%s', $this->thumb, $this->sizes['width']);
277:         
278:             //Sets the thumbnail path
279:             $this->thumb = sprintf('%s.jpg', $this->thumb);
280:         }
281:         //Else, if the mime type is not known
282:         else
283:             throw new InternalErrorException(__d('me_tools', 'The mime type {0} is not supported', $mime));
284:         
285:         //If the thumbnail does not yet exist
286:         if(!is_readable($this->thumb)) {
287:             //Checks if the target directory is writable
288:             if(!is_writable(dirname($this->thumb)))
289:                 throw new InternalErrorException(__d('me_tools', 'The target directory {0} is not writable', dirname($this->thumb)));
290:             
291:             //Creates the thumbnail
292:             if(preg_match('/image\/\S+/', $mime))
293:                 $this->_imageThumb();
294:             else
295:                 $this->_videoThumb();
296:         }
297:         
298:         //Renders the thumbnail
299:         header('Content-type: image/jpeg');
300:         readfile($this->thumb);
301:         
302:         $this->autoRender = FALSE;
303:         exit;
304:     }
305: }
