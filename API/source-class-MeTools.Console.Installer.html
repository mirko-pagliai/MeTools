 42:      * Assets for which create symbolic links.
 43:      * The key must be relative to `vendor/`, the value must be relative to `webroot/vendor/`

 55:     /**
 56:      * Creates a symbolic link to vendor asset.
 57:      * 
 58:      * For example:
 59:      * <pre>createSymbolicLinkToVendor('components/jquery', 'jquery');</pre>
 60:      * will create a symbolic link from `vendor/components/jquery` to `webroot/vendor/jquery`.
 61:      * @param string $from Name, relative to `vendor/`
 62:      * @param string $to Name, relative to `webroot/vendor/`
 63:      * @param \Composer\Script\Event $event The composer event object
 64:      */
 65:     public static function createSymbolicLinkToVendor($from, $to, $event) {
 66:         $io = $event->getIO();
 67:         
 68:         //Creates the target directory (`webroot/vendor/`)
 69:         if(!file_exists($webrootDir = ROOT.DS.'webroot'.DS.'vendor') && mkdir($webrootDir))
 70:             $io->write(sprintf('Created `%s` directory', str_replace(ROOT, NULL, $webrootDir)));
 71:         
 72:         //Deletes the link, if it already exists
 73:         if(file_exists($to = $webrootDir.DS.$to))
 74:             unlink($to);
 75:         
 76:         $from = $event->getComposer()->getConfig()->get('vendor-dir').DS.$from;
 77:                 
 78:         //Creates the symbolic link
 79:         if(@symlink($from, $to))
 80:             $io->write(sprintf('Created symbolic link to `%s`', str_replace(ROOT, NULL, $to)));
 81: 
 82:         else
 83:             $io->write(sprintf('Failed to create a symbolic link to `%s`', str_replace(ROOT, NULL, $to)));
 84:     }

110:     /**
111:      * Occurs after the autoloader has been dumped, either during install/update, or via the dump-autoload command.
112:      * @param \Composer\Script\Event $event The composer event object
113:      * @uses linksToAssets
114:      * @uses createSymbolicLinkToVendor()
115:      * @uses createWritableDirectories()
116:      * @uses setFolderPermissions()
117:      * @see https://getcomposer.org/doc/articles/scripts.md
118:      * @todo Needs to check for root/sudo
119:      */
120:     public static function postAutoloadDump(Event $event) {
121:         $io = $event->getIO();
122:         
123:         //Creates some (writable) directories
124:         static::createWritableDirectories(ROOT, $io);
125:         
126:         //If the shell is interactive
127:         if($io->isInteractive()) {
128:             $validator = function($arg) {
129:                 if(in_array($arg, ['Y', 'y', 'N', 'n']))
130:                     return $arg;
131:                 
132:                 throw new Exception('This is not a valid answer. Please choose Y or n.');
133:             };
134:             
135:             //Asks if the permissions should be changed
136:             $setFolderPermissions = $io->askAndValidate('<info>Set folder permissions? (Default to Y)</info> [<comment>Y, n</comment>]? ', $validator, 10, 'Y');
137: 
138:             if(in_array($setFolderPermissions, ['Y', 'y']))
139:                 self::setFolderPermissions(ROOT, $io);
140:             
141:             //Asks if the symbolic links to vendors should be created
142:             $createSymbolicLinkToVendor = $io->askAndValidate('<info>Create symbolic links to vendors? (Default to Y)</info> [<comment>Y, n</comment>]? ', $validator, 10, 'Y');
143:             
144:             if(in_array($createSymbolicLinkToVendor, ['Y', 'y']))
145:                 foreach(self::$linksToAssets as $from => $to)
146:                     self::createSymbolicLinkToVendor($from, $to, $event);
147:         }
148:         else {
149:             self::setFolderPermissions(ROOT, $io);
150: 
151:             foreach(self::$linksToAssets as $from => $to)
152:                 self::createSymbolicLinkToVendor($from, $to, $event);
153:         }
154:     }

156:     /**
157:      * Set globally writable permissions on some directories
158:      * @param string $dir The application's root directory
159:      * @param \Composer\IO\IOInterface $io IO interface to write to console
160:      */
161:     public static function setFolderPermissions($dir, $io) {
162:         //Change the permissions on a path and output the results
163:         $changePerms = function ($path, $perms, $io) {
164:             //Get current permissions in decimal format so we can bitmask it
165:             $currentPerms = octdec(substr(sprintf('%o', fileperms($path)), -4));
166:             if(($currentPerms & $perms) == $perms)
167:                 return;
168:             
169:             $res = chmod($path, $currentPerms | $perms);
170:             if($res)
171:                 $io->write('Permissions set on '.$path);
172:             else
173:                 $io->write('Failed to set permissions on '.$path);
174:         };
175: 
176:         $walker = function ($dir, $perms, $io) use (&$walker, $changePerms) {
177:             $files = array_diff(scandir($dir), ['.', '..']);
178:             foreach ($files as $file) {
179:                 $path = $dir.'/'.$file;
180: 
181:                 if (!is_dir($path))
182:                     continue;
183: 
184:                 $changePerms($path, $perms, $io);
185:                 $walker($path, $perms, $io);
186:             }
187:         };
188: 
189:         $worldWritable = bindec('0000000111');
190:         $walker($dir.'/tmp', $worldWritable, $io);
191:         $changePerms($dir.'/tmp', $worldWritable, $io);
192:         $changePerms($dir.'/logs', $worldWritable, $io);
193:         $changePerms($dir.'/webroot/files', $worldWritable, $io);
194:     }
